### **🔧 Новые методы:**
```typescript
// Форматирование даты по-русски
const formatDate = (date: Date) => {
  const options: Intl.DateTimeFormatOptions = {
    weekday: 'long',
    day: 'numeric', 
    month: 'long',
    year: 'numeric'
  };
  return date.toLocaleDateString('ru-RU', options);
};

// Переход в настройки (уже был)
const goToSettings = () => {
  router.push('/settings');
};
```
